func prKv(s: [String:AnyObject],buf:String = "") {
    for i in s.keys {
        if let t = s[i] as? [String: AnyObject] {
            print(buf,i)
            let tbuf=buf+"-"
            prKv(t,buf: tbuf)
        } else {
            print(buf,i," ",s[i]!)
        }
    }
}
func buildD(s: [String:AnyObject],inout u:[String:AnyObject] ) -> [String:AnyObject] {
    for i in s.keys {
        if let t = s[i] as? [String: AnyObject] {
            u[i]=buildD(t,u: &u)
        } else {
            u[i]=s[i]
        }
    }
    return u
}
