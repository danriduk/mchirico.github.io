func lastDay(month: Int, year: Int) -> (Int,String)? {
  let calendar = NSCalendar.currentCalendar()
  let comps = NSDateComponents()
  comps.month = month+1
  comps.year = year
  comps.day = 0
  calendar.dateFromComponents(comps)
  
  guard let lastDay = calendar.dateFromComponents(comps) else {
    return nil
  }
  
  // Monday/Tuesday
  let dfmt = NSDateFormatter()
  dfmt.dateFormat = "EEEE"
  let dateString = dfmt.stringFromDate(lastDay)
  
  let tag = calendar.components([.WeekOfMonth,.WeekOfYear,
    .YearForWeekOfYear,.Weekday],    fromDate: lastDay)
    
  return (tag.weekday,dateString) //Saturday == 0
}

extension String {
  
  func regex (pattern: String) -> [String] {
    do {
      let regex = try NSRegularExpression(pattern: pattern, options: NSRegularExpressionOptions(rawValue: 0))
      let nsstr = self as NSString
      let all = NSRange(location: 0, length: nsstr.length)
      var matches : [String] = [String]()
      regex.enumerateMatchesInString(self, options: NSMatchingOptions(rawValue: 0), range: all) {
        (result : NSTextCheckingResult?, _, _) in
        
        if let r = result {
          let result = nsstr.substringWithRange(r.range) as String
          matches.append(result)
        }
      }
      return matches
    } catch {
      return [String]()
    }
  }
  
}

func findAndNext(s0: String, s1: String,
  var mr:Range<String.CharacterView.Index>? = nil) -> [Range<String.CharacterView.Index>?] {
    var aar:[Range<String.CharacterView.Index>?] = []
    
    if let t = s0.rangeOfString(s1, range: mr) {
      aar.append(t)
      mr = t
      mr?.startIndex = mr!.startIndex.advancedBy(1)
      mr?.endIndex = s0.endIndex
      aar = aar + findAndNext(s0,s1: s1,mr: mr)
    }
    
    return aar
}
